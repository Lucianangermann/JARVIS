# --- Communication layer ---
# No new hard deps: messaging (Telegram Bot API) uses requests; social
# (Reddit RSS) uses httpx; translation reuses the Claude client. iMessage /
# calls / email go through macOS AppleScript + chat.db (stdlib sqlite3).

# Voice (STT + TTS) is optional — install separately if needed:
#     pip install -r requirements-voice.txt
# It pulls torch + llvmlite which can fail to build on some macOS setups.

# --- Memory + self-learning ---
# Local vector store + embedding model. ChromaDB persists under
# data/chromadb/, sentence-transformers caches model weights under
# ~/.cache/huggingface. transformers is pinned because the newest
# (5.x) requires PyTorch ≥ 2.4 — Intel macOS wheels stopped at 2.2,
# so we pin to 4.x to stay compatible.
chromadb>=0.5
sentence-transformers>=2.7
transformers<4.50
scikit-learn>=1.4
# numpy MUST stay <2.0 because PyTorch (transitive via sentence-
# transformers) on Intel macOS pinned at 2.2 has no numpy-2.x ABI.
# Bumping numpy past 2.0 silently breaks long-term embeddings.
numpy>=1.26,<2.0

# --- Vision (Phase 1) ---
# Screen capture goes through mss (Cocoa CGWindowList wrapper on macOS,
# X11/Win32 elsewhere) which is faster and more portable than mucking
# with PyObjC directly. Pillow handles JPEG encoding + resizing for the
# Claude Vision payload — every image is re-encoded to JPEG ≤1920px to
# keep token cost and request size bounded.
mss>=9.0
# Pillow is already pulled in transitively by sentence-transformers /
# chromadb, but list it explicitly so vision keeps working if those
# memory deps ever change.
Pillow>=10.0

# --- Vision (Phase 3) ---
# OpenCV for camera capture + frame-diff motion detection. Pinned
# <4.11 because the 4.12 wheels require numpy 2.x, which silently
# breaks the older PyTorch / sentence-transformers C extensions
# the memory layer depends on (long-term embeddings won't load).
# numpy is pinned <2.0 in lockstep — keep both in sync if you bump.
opencv-python<4.11
# pyzbar is optional for barcode/QR decoding (Phase 2). The Python
# package is small but needs the native libzbar — on macOS install
# via `brew install zbar`. We don't list it as a hard dep because
# barcode scanning gracefully no-ops if it's missing.

# --- Security & monitoring layer ---
# psutil powers system_monitor (CPU/RAM/disk/battery/network) and the
# open-port scan in digital_security. Pure-Python with a small C
# extension; builds cleanly everywhere. Zero-config — the moment it's
# installed, /security/system/* works.
psutil>=5.9
# bcrypt hashes the fallback PIN for voice_auth (never stored plaintext).
bcrypt>=4.1
# cryptography (Fernet) encrypts sensitive at-rest DB content (message bodies,
# expense notes) when JARVIS_DB_KEY is set — see server/common/crypto.py.
cryptography>=42.0
# Speaker verification (resemblyzer) is OPTIONAL and heavier — it pulls
# torch + librosa. Install it only if you want voice authentication via
# requirements-voice.txt. voice_auth degrades to PIN-only without it.
